```python
import spritePro as s
from tools.sprite_editor.scene import Scene

# –ó–∞–≥—Ä—É–∂–∞–µ–º —Å—Ü–µ–Ω—É
scene = Scene.load("level1.json")

# –°–æ–∑–¥–∞—ë–º –∏–≥—Ä–æ–≤—ã–µ —Å–ø—Ä–∞–π—Ç—ã
s.get_screen((800, 600), "My Game")

sprites_map = {}  # id -> s.Sprite

for obj in scene.objects:
    if not obj.visible:
        continue
    
    sprite = s.Sprite(
        obj.sprite_path,
        (64, 64),  # —Ä–∞–∑–º–µ—Ä (–º–æ–∂–Ω–æ –ø–æ–ª—É—á–∏—Ç—å –∏–∑ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è)
        (obj.transform.x, obj.transform.y),
        scene=s.get_current_scene()
    )
    
    # –ü—Ä–∏–º–µ–Ω—è–µ–º —Ç—Ä–∞–Ω—Å—Ñ–æ—Ä–º–∞—Ü–∏–∏
    if obj.transform.rotation:
        sprite.angle = obj.transform.rotation
    if obj.transform.scale_x != 1:
        sprite.scale = obj.transform.scale_x
    
    # Z-index —á–µ—Ä–µ–∑ sorting_order
    sprite.sorting_order = obj.z_index
    
    sprites_map[obj.id] = sprite

# –ó–∞–ø—É—Å–∫–∞–µ–º –∏–≥—Ä—É
while True:
    s.update(fill_color=(20, 20, 30))
```
